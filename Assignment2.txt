1.SELECT D.DNAME,DEPTNO
FROM DEPT D
WHERE NOT EXISTS 
		(SELECT *
		FROM EMP E
		WHERE D.DEPTNO=E.DEPTNO);

2.SELECT ENAME,JOB
FROM EMP 
WHERE MGR IS NULL;

3.SELECT E.ENAME,E.JOB,D.LOC
FROM EMP E, DEPT D
WHERE E.DEPTNO=D.DEPTNO AND COMM>1000 AND LOC LIKE 'NEW YORK'

4.SELECT COUNT(E.EMPNO) AS 'TOTAL EMP'
FROM EMP E, DEPT D
WHERE E.DEPTNO=D.DEPTNO AND LOC LIKE 'CHICAGO';

5.SELECT ENAME,EMPNO,JOB,LOC
FROM EMP E, DEPT D
WHERE E.DEPTNO=D.DEPTNO AND LOC LIKE 'CHICAGO';

6.SELECT ENAME,LOC
FROM EMP E, DEPT D
WHERE E.DEPTNO=D.DEPTNO 
ORDER BY LOC;

7.SELECT COUNT(ENAME) AS 'NUM OF EMP',COUNT(DISTINCT DEPTNO) AS 'NUM OF DEPT'
FROM EMP E;

8.SELECT E.ENAME,M.ENAME AS 'MANAGER', M.HIREDATE AS 'MANAGER HIREDATE',E.HIREDATE
FROM EMP E, EMP M
WHERE E.MGR=M.EMPNO AND E.HIREDATE<M.HIREDATE;

9.SELECT COUNT(ENAME) AS 'NUM OF EMP'
FROM EMP E
WHERE JOB LIKE 'MANAGER';

10.SELECT ENAME,JOB
FROM EMP 
WHERE MGR IS NULL;

11.SELECT ENAME,DEPTNO
FROM EMP 
WHERE ENAME LIKE 'SMITH';

12.SELECT E.ENAME, E.JOB, D.LOC 
FROM EMP E, DEPT D
WHERE E.DEPTNO=D.DEPTNO AND JOB LIKE 'SALESMAN' AND LOC NOT LIKE 'BOSTON';

13.SELECT E.*, D.DNAME,D.LOC 
FROM EMP E, DEPT D
WHERE E.DEPTNO=D.DEPTNO;

14.SELECT ENAME, JOB
FROM EMP
WHERE JOB LIKE 'SALESMAN' AND SAL<
        (SELECT SAL 
        FROM EMP
        WHERE ENAME='MILLER');

15.SELECT  E.*, D.LOC
FROM EMP E, DEPT D
WHERE E.DEPTNO=D.DEPTNO and D.LOC LIKE 'CHICAGO';

16.SELECT ENAME
FROM EMP
WHERE SAL >
           (SELECT MIN(SAL) 
           FROM EMP 
           WHERE JOB LIKE 'MANAGER');

17.SELECT E.ENAME,E.EMPNO
FROM EMP E
WHERE E.DEPTNO =
           (SELECT D.DEPTNO 
           FROM DEPT D
           WHERE DNAME LIKE 'ACCOUNTING');

18.SELECT ENAME,EMPNO,JOB
FROM EMP
WHERE JOB LIKE 'MANAGER';

19.SELECT DNAME,
CASE
    WHEN DNAME='SALES' THEN 'SALES_DEPT'
    ELSE 'NON_SALES'
END AS DEPT_NAME
FROM EMP 
JOIN DEPT on EMP.DEPTNO=DEPT.DEPTNO;

20.SELECT ENAME,SAL,
CASE
    WHEN SAL>1000 THEN SAL
ELSE 1000
END AS NEW_SAL
FROM EMP ;

21.SELECT SAL,
CASE
    WHEN SAL<1000 THEN 'LESS'
WHEN SAL BETWEEN 1000  and 2500 THEN 'MEDIUM'
ELSE 'HIGH'
END AS SAL_STATUS
FROM EMP;
